def reverse_double(input:list)->list:
  for i in range(len(input)):
    input[i] = input[i] * 2

  input.reverse()
  return input

def power_iter(X, num_iter:int, rng = np.random.RandomState(random_seed)):

  v = rng.randn(X.shape[1])  # Initialize with random vector with shape (d,)
  one_vec = np.ones(X.shape[0])  # All ones vector with shape (n,)       is the 1 from instructions
  mu_row_matrix = np.mean(X, axis=0)  # Returns a 1 row matrix with shape (d, 1) since X is of "matrix" type
  mu = np.array(mu_row_matrix).squeeze()  # Convert from a sparse column matrix to a dense 1D array with shape (d,)    is the u from instructions

  ######  <YOUR CODE>  ###### ######

  # Xc = X - np.mean(X, axis=0)     # .T performs transpose on the matrix
  X_trans = X.T
  one_vec_trans = one_vec.T
  mu_trans = mu.T


  for i in range(1, num_iter):
    v_temp = X_trans.dot(X.dot(v))
    v_temp = v_temp - mu.dot(one_vec_trans.dot(X.dot(v)))
    v_temp = v_temp - (X_trans.dot(one_vec)).dot(mu_trans.dot(v))
    # print(type(one_vec_trans.dot(one_vec)))
    v_temp = v_temp + mu * ((one_vec_trans.dot(one_vec)) * (mu_trans.dot(v)))     # one_vec_trans.dot(one_vec)   returns a float so I can't use .dot on it
    v = v_temp
    v = v / (np.linalg.norm(v))


  ###### ###### ###### ###### ######
  return v
